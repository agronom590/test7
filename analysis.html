def time_gen (order_conseq):
    random.seed()
    x = []
    for i in range(0,len(order_conseq)): 
        x = x + [((round(random.uniform(0,8),1)) + 8 * i) for j in range(0, order_conseq[i])] #        - 
    return sorted(x)  #

def experiment (val, disp, period, staff):
    order_conseq = order_gen(val=val, disp=disp, period=period) #   
    times = time_gen (order_conseq) #    

    #  , ,     ,   ,     
    #       16 
    #     

    #  ,    ,          
    #       ,   
    busy = [0 for i in range(0, staff)]

    #       
    decision_time = [0 for i in range(0, len(times))]
    
    for i in range(0, len(times)):
        if times[i] >= min(busy):
            decision_time[i] = times[i] + 3    #      -     
        else:
            decision_time[i] = min(busy) + 3   #    -     
        busy[busy.index((min(busy)))] = decision_time[i]   #     
    
    
    #  ,           16  (2  )
    res_time = [(decision_time[i] - times[i]) for i in range(0,len(times))]  #     
    res = [1 if i <= 16 else 0 for i in res_time]   # : 1 -   , 0 - 
    
    print('  : ', period)
    print(' : ', staff)
    print('     ( ): '+str(round(statistics.mean(order_conseq),1))+' ('+
         str(round(statistics.stdev(order_conseq),1))+')')
    print('    ( ),  : '+str(round(statistics.mean(res_time),1))+' ('+
         str(round(statistics.stdev(res_time),1))+')')
    print('     16    : '+str(round(sum(res)/len(res)*100,1)))
    
    rcParams['figure.figsize'] = 10, 8
